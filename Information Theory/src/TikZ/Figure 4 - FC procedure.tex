\begin{document}
    \begin{procedure}[!h]
        \caption{buildFC()}\label{proc:pseudoFC} 
        $fc[MAX] = 0$ \DontPrintSemicolon\;
        \For{$l \leftarrow MAX - 1$ \KwTo 1}{
            $fc[l] \leftarrow (fc[l + 1] + num[l + 1]) << 1$\DontPrintSemicolon\;
            $l \leftarrow l - 1 $
        }
    \end{procedure}
\end{document}
