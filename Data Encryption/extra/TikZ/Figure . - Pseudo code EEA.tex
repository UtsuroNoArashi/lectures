    \begin{algorithm}
        \caption{Extended Euclidean Algorithm}
        \KwIn{two non-negative Integers \(a, b\) with \(a \ge b\).}
        \KwOut{\(d = \gcd{a, b}\) and the integers \(x, y\) such that \(ax + by = d\).}
        If \(b = 0\), return \((a, 1, 0)\).
        \DontPrintSemicolon\;
        Set \(x_{2} \leftarrow 1, x_{1} \leftarrow 0, y_{2} \leftarrow 0, y_{1} \leftarrow 1\).
        \DontPrintSemicolon\;
        While \(b > 0\) do:
            \DontPrintSemicolon\;
            \Indp\(
                    q \leftarrow \floor{\sfrac{a}{b}}, a \leftarrow a - qb,
                        x \leftarrow x_{2} - qx_{1}, y \leftarrow y_{2} - qx_{1}
                \).
            \DontPrintSemicolon\;
            \(
                    a \leftarrow b, b \leftarrow r, x_{2} \leftarrow x_{1} 
                        x_{1} \leftarrow x, y_{2} \leftarrow y_{1},
                        \text{ and } y_{1} \leftarrow y
                \).
            \DontPrintSemicolon\;
        \Indm Return \((a, x_{2}, y_{2})\).
    \end{algorithm}
