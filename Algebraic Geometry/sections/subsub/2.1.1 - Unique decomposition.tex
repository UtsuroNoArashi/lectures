    \begin{definition*}
        Let \(X\) be a topological space. We say that \(X\) admits a decomposition
            in inreducible closed sets, if there exists \(X_{1}, \ldots, X_{n}\)
            inreducible closed sets, such that \(X = X_{1} \cup \cdots \cup X_{n}\).
    \end{definition*}

    \begin{lemma*}
        Let \(X\) be a topological space. We say that a decomposition 
            \(X = X_{1} \cup \cdots \cup X_{n}\) is irredundant if any of the following holds.
            \begin{enumerate}
                \item \label{itm:1} For all \(i\), it holds \(X_{i} \not\subset \bigcup_{j \ne i} X_{j}\).
                \item \label{itm:2} \(\forall i \ne j, X_{i} \not\subset X_{j}\).
            \end{enumerate}
    \end{lemma*}

    \begin{proof*}
        We prove that the two statements are equivalent.
        \begin{description}
            \item [\(\ref{itm:1} \implies \ref{itm:2}\):] By the sake of contradiction, let \ref{itm:2} be false.
                Then, there exists \(i\) such that \(X_{i} \subseteq X_{j}\), 
                meaning 
                \[
                    X_{i} \subseteq \bigcup_{j \ne i} X_{j}.
                \]
                Which contradicts the hypotesis.

            \item [\(\ref{itm:2} \implies \ref{itm:1}\):] By a similar reasoning as above, 
                if \ref{itm:1} does not hold, then there exists \(i\) such that 
                \(X_{i} \subseteq \bigcup_{j \ne i} X_{j}\).

                Hence, \(X_{i} = \bigcup_{j \ne i} (X_{j} \cap X_{i}) = \bigcup_{j \ne i} Y_{j}\).
                Each \(Y_{i}\) is closed in \(X_{i}\) but not proper since the \(X_{i}\)
                    are inreducible.
                Therefore, there exists \(j \ne i\) such that \(Y_{j} = X_{i}\);
                that is, \(X_{i} \cap X_{j} = X_{i} \iff X_{i} \subseteq X_{j}\).\qed
        \end{description}
    \end{proof*}
    
    \begin{theorem*}
        Let \(X\) be a Noetherian space. Then, \(X\) admits an irredundant decomposition 
            in inreducible closed sets. Additionally, such decomposition is unique 
            apart from the ordering.
    \end{theorem*}

    \begin{proof*}
        We divide the proof in three parts: we first show that a decomposition exists,
        then we prove that it's irredundant and lastly we show its uniqueness.

        \begin{enumerate}
            \item Let \(\F*\) be a family of closed sets in \(X\) that admit no decomposition.
                We show that \(\F* = \varnothing\).

                By the sake of contradiction assume \(\F \ne \varnothing\).
                Then there exists \(Z \in \F*\), such that \(Z\) is minimal.
                We have that
                \begin{enumerate}
                    \item If \(Z\) is inreducible, 
                        then \(Z = Z\) is a decomposition in inreducible closed sets. 
                        Which goes against the hypotesis.

                    \item If \(Z\) is reducible, then we can write \(Z = Z_{1} \cup \cdots Z_{k}\),
                        with each \(Z_{i}\) a proper closed set. Hence, \(Z_{i} \ne \F*\);
                        that is, \(Z_{i}\) admits a decomposition, which is again a contradiction.
                \end{enumerate}

            \item Trivial. Let \(X = X_{1} \cup \cdots \cup X_{k}\) be a decomposition with a minimal element.
                By the above lemma, \(\forall i, X_{i} \not\subset \bigcup_{j \ne i }X_{j}\).

            \item Let \(X = X_{1} \cup X_{2} \cup \cdots \cup X_{k}\) and 
                \(X = Y_{1} \cup Y_{2} \cup \cdots \cup Y_{l}\) be two decomposition for \(X\),
                with \(X_{i}, Y_{j}\) inreducible closed sets. 
                We have 
                \[\begin{aligned}
                    X_{i} &= X_{i} \cap X \\ 
                        &= X_{i} \cap (Y_{1} \cup Y_{2} \cap \cdots \cap Y_{l}) \\ 
                        &= \bigcup_{j \ne i} (Y_{j} \cap X_{i}).
                \end{aligned}\]

                But we said that each \(X_{i}\) is inreducible, hence \(\exists j = j(i)\),
                where by \(j(i)\) that \(j \text{ depends from } i\), such that \(X_{i} \cap Y_{j(i)} = X_{i}\);
                that is, such that \(X_{i} \subseteq Y_{j(i)}\).

                Applying the same reasoning for the \(Y_{j}\)s, we get 
                \[
                    X_{i} \subseteq Y_{j(i)} \subseteq X_{i(j(i))}.
                \]
                Hence, \(X_{i} = Y_{j(i)}\).\qed
        \end{enumerate}
    \end{proof*}
